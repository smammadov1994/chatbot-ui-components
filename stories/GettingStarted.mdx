## Installation

```bash
npm install @pumpads/chat-component-library
```

or

```bash
yarn add @pumpads/chat-component-library
```

## Basic Usage

```tsx
import { ChatInterface } from '@pumpads/chat-component-library'
import '@pumpads/chat-component-library/styles'
```

```tsx
import { useState } from 'react'
import { ChatInterface, ChatMessage } from '@pumpads/chat-component-library'

function App() {
  const [messages, setMessages] = useState<ChatMessage[]>([])

  const handleSend = (content: string) => {
    const newMessage: ChatMessage = {
      id: Date.now().toString(),
      role: 'user',
      content,
      timestamp: new Date()
    }
    setMessages([...messages, newMessage])
  }

  return (
    <ChatInterface
      messages={messages}
      onSendMessage={handleSend}
      theme="dark"
    />
  )
}
```

## Theme Support

```tsx
<ChatInterface messages={messages} theme="dark" />
```

## Message Types

```tsx
const userMessage = { id: '1', role: 'user', content: 'Hello!', timestamp: new Date() }
const assistantMessage = {
  id: '2',
  role: 'assistant',
  content: 'Hi there! How can I help?',
  timestamp: new Date(),
  thoughtTime: 2
}
```

## Markdown Support

> Assistant messages support **Markdown** automatically:
>
> ```
> **Bold text**
> *Italic text*
> - List item
> ```

## Loading States

```tsx
const [isLoading, setIsLoading] = useState(false)

<ChatInterface
  messages={messages}
  isLoading={isLoading}
  onSendMessage={handleSend}
/>
```

## Customization

```css
:root {
  --accent-primary: #4F46E5;
  --background-primary: #0a0a0a;
  --text-primary: #fcfcfc;
}
```

## Message Actions

```tsx
<ChatInterface
  messages={messages}
  onMessageAction={(action, messageId) => {
    console.log(`Action ${action.type} on message ${messageId}`)
  }}
/>
```

## Custom Thinking Phrases

```tsx
<ChatInterface
  messages={messages}
  thinkingPhrases={[
    'Processing your request...',
    'Analyzing data...',
    'Generating response...'
  ]}
/>
```
